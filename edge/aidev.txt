import { DefaultRequestHeaders } from "./configs";

import { ShareLinkParser } from "./Parsers/share-link";
import { ClashMetaParser } from "./Parsers/clash-meta";

import Yaml from "js-yaml";
import { TrulyAssign } from "./utils/TrulyAssign";

type SubURL = string; // start with `short:`, `http(s)://`, or sth like `[proxy protocol]://`
type SubURLs = string; // contains SubURL a lot
type SubURLArr = SubURL[]; // usually it is some SubURLs squeeze into a Array.

/**
 * 
 * @returns {Array}
 */
export default async function getParsedSubData (
    SubURLs: SubURLs, 
    EdgeSubDB, 
    isShowHost = false as boolean,
    CustomHTTPHeaders = {} as Headers
) {
    let __startTime = performance.now();
    console.info("[Fetch Sub Data] Job started")
    console.info(`[Fetch Sub Data] isShowHost: ${isShowHost.toString()}`)

    const RequestHeaders = TrulyAssign(DefaultRequestHeaders, CustomHTTPHeaders);

    let SubURLArr = SubURLs.replaceAll("\r", "\n").split("\n").filter((i) => !!i).map(i => encodeURIComponent(i.trim())).map(i => decodeURIComponent(i)) as SubURLArr;
    let ParsedData = [];
    for (let i in SubURLArr) {
        console.info(`[Fetch Sub Data] Fetching ${parseInt(i) + 1}/${SubURLArr.length}`)
        ParsedData = [...ParsedData, ...(await ParseSubData(SubURLArr[i], EdgeSubDB, RequestHeaders))]
    }

    if (isShowHost === true) {
        ParsedData = ParsedData.map(i => {
            i.__Remark = `${i.__Remark} - ${i.Hostname}:${i.Port}`
            return i;
        })
    }

    console.info(`[Fetch Sub Data] Job done, wasting ${performance.now() - __startTime}ms.`)
    return ParsedData;
}
async function ParseSubData (SubURL: SubURL, EdgeSubDB, RequestHeaders) {
    // handle `short` here
    if (SubURL.match(/^short:/i)) {
        let ShortData: SubURLs = await EdgeSubDB.get(SubURL).then(res => JSON.parse(res).subdata);
        console.info(`[Fetch Sub Data] starting sub task for ${SubURL}`)
        let ParsedShortData = await getParsedSubData(ShortData, EdgeSubDB, RequestHeaders);
        console.info(`[Fetch Sub Data] sub task for ${SubURL} done`)
        return ParsedShortData;
    }

    let SubData;
    // handle bundle share link (not starts with `http(s|)://` )
    if (!SubURL.match(/^http(s|):\/\//i)) {
        SubData = {
            type: "share-link",
            data: SubURL
        }
    } else {
        SubData = await fetch(SubURL, { headers: RequestHeaders })
            .then(async res => await res.text())
            .then(res => {
                // try decode as yaml, for clash-meta config
                try {
                    let YamlData = Yaml.load(res) as ClashMetaConfig;
                    if (YamlData.proxies) {
                        return {
                            type: "clash-meta",
                            data: YamlData
                        }
                    }
                } catch (e) {
                    console.warn(`[Fetch Sub Data] Err when try parse as yaml: ${e}, skipping...`)
                 }
                 // try decode as json, for sing-box config
                 try {
                     let SingBoxConfig = JSON.parse(res);
                     if (SingBoxConfig.outbounds) {
                         return {
                             type: "sing-box",
                             data: SingBoxConfig
                         }
                     }
                 } catch (e) {
                     console.warn(`[Fetch Sub Data] Err when try parse as JSON (sing-box): ${e}, skipping...`)
                }

                // try decode as base64 endoded share-links
                try {
                    let decodedData = atob(res.trim());
                    if (!decodedData.match(/\:\/\//gi)) {
                        throw "seems like base64 decoded data malformed"
                    }
                    return {
                        type: "share-link",
                        data: decodedData
                    }
                } catch (e) {
                    console.warn(`[Fetch Sub Data] try parse as base64: ${e}, skipping...`)
                }

                // final return as share-link
                return {
                    type: "share-link",
                    data: res
                }
            })
    }

    let ParsedSubData = [];
    if (SubData.type === "share-link") {
        let links = SubData.data.replaceAll("\r", "\n").split("\n").filter(loc => !!loc);
        let Parser = new ShareLinkParser();
        for (let i of links) {
            let protocol = i.split(":")[0];
            try {
                if (Parser.__validate(i)) {
                    ParsedSubData.push(Parser[protocol](i))
                }
            } catch (e) {
                console.warn(`[Fetch Sub Data] this share-url doesn't seem right, ignoring... ('${i}')`)
                console.warn(e)
            }
        }
    } else if (SubData.type === "clash-meta") {
        let { proxies } = SubData.data;
        let Parser = new ClashMetaParser();
        for (let i of proxies) {
            if (Parser.__validate(i)) {
                ParsedSubData.push(Parser[i.type](i))
            }
        }
    } else if (SubData.type === "sing-box") {
         let { outbounds } = SubData.data;
         let Parser = new SingBoxParser();
         for (let i of outbounds) {
             if (Parser.__validate(i)) {
                 ParsedSubData.push(Parser[i.type](i))
             }
         }
    }
    
    return ParsedSubData;
}

export async function getClashMetaConfig(
    Proxies,
    EdgeSubDB,
    PassedConfig = {},
) {
    const Config = TrulyAssign(BasicConfig, PassedConfig);

    let RemoteProxyGroups = [];
    try {
        const RemoteReaderResult = await (new RemoteConfigReader(Config.RemoteConfig)).Process(EdgeSubDB, Config.isForcedRefresh);
        RemoteProxyGroups = Array.isArray(RemoteReaderResult?.ProxyGroup) ? RemoteReaderResult.ProxyGroup : [];
        if (RemoteProxyGroups.length > 0) {
            console.info("[getClashMetaConfig] Successfully processed proxy groups from remote config.");
        } else {
            console.info("[getClashMetaConfig] No proxy groups found or processed from remote config.");
        }
    } catch (e) {
        console.error("[getClashMetaConfig] Error processing remote config for proxy groups:", e);
        RemoteProxyGroups = [];
    }

    let ClashConfig = JSON.parse(JSON.stringify(BasicClashConfig));
    let Dumper = new ClashMetaDumper(Config.isUDP, Config.isSSUoT, Config.isInsecure);

    Proxies = Proxies.map(i => {
        if (Dumper.__validate(i)) {
            if (typeof i.Hostname === 'string') {
                i.Hostname = i.Hostname.replace(/(^|$)/g, "");
            }
            return i;
        }
    }).filter(i => !!i);
    ClashConfig.proxies = Proxies.map(i => Dumper[i.__Type](i));
    console.info("[getClashMetaConfig] Available proxies:", Proxies.map(p => p.__Remark));

    const existingGroupNames = new Set(ClashConfig["proxy-groups"].map(group => group.name));

    for (let i of RemoteProxyGroups) {
        let MatchedProxies = [];
        if (Array.isArray(i.RegExps)) {
            for (let t of i.RegExps) {
                try {
                    if (t && typeof t === 'string') {
                        try {
                            const regex = new RegExp(t);
                            MatchedProxies = [...MatchedProxies, ...Proxies.filter(loc => loc?.__Remark?.match(regex))];
                        } catch (reError) {
                            console.warn(`[getClashMetaConfig] Invalid RegExp pattern "${t}" in group "${i?.name}". Skipping. Error: ${reError?.message}`);
                        }
                    }
                } catch (e) {
                    console.warn(`[getClashMetaConfig] Error filtering proxies with regex "${t}" in group "${i?.name}". Skipping. Error: ${e?.message}`);
                }
            }
        }
        console.info(`[getClashMetaConfig] Matched proxies for group "${i?.name}" with regex "${i.RegExps}":`, MatchedProxies.map(p => p.__Remark));

        const uniqueProxyNames = new Set(MatchedProxies.map(p => p.__Remark));
        const uniqueMatchedProxies = Proxies.filter(p => uniqueProxyNames.has(p.__Remark));
        console.info(`[getClashMetaConfig] Unique matched proxies for group "${i?.name}":`, uniqueMatchedProxies.map(p => p.__Remark));

        let GroupProxies = [];
        const testConfigTypes = ["url-test", "fallback", "load-balance"];

        if (testConfigTypes.includes(i.type)) {
            GroupProxies.push(...uniqueMatchedProxies.map(t => t.__Remark));
        }

        if (Array.isArray(i.GroupSelectors)) {
            GroupProxies.push(...i.GroupSelectors.map(t => {
                if (typeof t === 'string') {
                    return t.replace(/\[|\]/g, "").trim();
                }
                return '';
            }).filter(Boolean));
        }

        if (!testConfigTypes.includes(i.type)) {
            GroupProxies.push(...uniqueMatchedProxies.map(t => t.__Remark));
        }

        console.info(`[getClashMetaConfig] Group "${i?.name}" proxies before fallback:`, GroupProxies);

        if (GroupProxies.length === 0) {
            console.warn(`[getClashMetaConfig] Proxy group "${i?.name}" has no proxies or selectors. Adding fallback: DIRECT.`);
            GroupProxies.push("DIRECT");
        }

        let ProxyGroup = {
            name: i.name || `UnnamedGroup_${ClashConfig["proxy-groups"].length}`,
            type: i.type,
            proxies: GroupProxies
        };

        if (i.TestConfig && typeof i.TestConfig === 'object' && testConfigTypes.includes(i.type)) {
            if (i.TestConfig.TestURL) ProxyGroup.url = i.TestConfig.TestURL;
            if (i.TestConfig.Interval !== undefined) ProxyGroup.interval = i.TestConfig.Interval;
            if (i.type === "url-test" && i.TestConfig.Tolerance !== undefined) {
                ProxyGroup.tolerance = i.TestConfig.Tolerance;
            }
        }

        console.info(`[getClashMetaConfig] Final group "${i?.name}":`, ProxyGroup);

        if (existingGroupNames.has(ProxyGroup.name)) {
            console.warn(`[getClashMetaConfig] Duplicate group name "${ProxyGroup.name}" found. Replacing existing group.`);
            const index = ClashConfig["proxy-groups"].findIndex(group => group.name === ProxyGroup.name);
            if (index !== -1) {
                ClashConfig["proxy-groups"][index] = ProxyGroup;
            }
        } else {
            ClashConfig["proxy-groups"].push(ProxyGroup);
            existingGroupNames.add(ProxyGroup.name);
        }
    }

    ClashConfig.rules = [];
    try {
        console.info(`[getClashMetaConfig] Fetching rules config from: ${Config.RemoteConfig}`);
        const rawINI = await fetchCached(Config.RemoteConfig, "RemoteConfig", EdgeSubDB, Config.isForcedRefresh);

        if (!rawINI || typeof rawINI !== 'string') {
            throw new Error("Failed to fetch remote rules config or it was empty/invalid.");
        }

        const lines = rawINI.split('\n');
        let finalRuleFound = false;

        for (const line of lines) {
            const trimmedLine = line.trim();
            if (trimmedLine.startsWith(';') || trimmedLine === '') continue;

            if (trimmedLine.toLowerCase().startsWith("ruleset=")) {
                const ruleDef = trimmedLine.substring("ruleset=".length);
                const ruleParts = ruleDef.split(',');
                
                if (ruleParts.length >= 2) {
                    const ruleSetName = ruleParts[0].trim();
                    let ruleTarget = ruleParts[1].trim();
                    
                    if (ruleSetName.toUpperCase() === "MATCH") {
                        ClashConfig.rules.push(`MATCH,${ruleTarget}`);
                        finalRuleFound = true;
                        continue;
                    }
            
                    if (!ClashConfig["rule-providers"] || !ClashConfig["rule-providers"][ruleSetName]) {
                        console.warn(`[getClashMetaConfig] Rule provider "${ruleSetName}" not defined. Skipping.`);
                        continue;
                    }
                    
                    let clashRule = `RULE-SET,${ruleSetName},${ruleTarget}`;
                    ClashConfig.rules.push(clashRule);
                } else {
                    console.warn(`[getClashMetaConfig] Skipping malformed ruleset line: ${trimmedLine}`);
                }
            } 
            else if (trimmedLine.includes(',')) {
                const ruleParts = trimmedLine.split(',');
                if (ruleParts.length >= 2) {
                    const ruleType = ruleParts[0].trim().toUpperCase();
                    const ruleTarget = ruleParts[1].trim();
                    const options = ruleParts.slice(2).map(opt => opt.trim());

                    if (ruleType === "MATCH" || ruleType === "FINAL") {
                        const clashRule = `MATCH,${ruleTarget}`;
                        ClashConfig.rules.push(clashRule);
                        finalRuleFound = true;
                    } else if (["DOMAIN-SUFFIX", "DOMAIN-KEYWORD", "DOMAIN", "IP-CIDR", "GEOIP", "SRC-IP-CIDR", "SRC-PORT", "DST-PORT", "PROCESS-NAME", "PROCESS-PATH"].includes(ruleType)) {
                        if (options[0]) {
                            let clashRule = `${ruleType},${ruleTarget},${options[0]}`;
                            if (options.length > 1 && options[1] === 'no-resolve') {
                                clashRule += ',no-resolve';
                            }
                            ClashConfig.rules.push(clashRule);
                        } else {
                            console.warn(`[getClashMetaConfig] Skipping rule line with missing target policy: ${trimmedLine}`);
                        }
                    } else {
                        console.warn(`[getClashMetaConfig] Skipping unrecognized rule line: ${trimmedLine}`);
                    }
                }
            }
        }

        if (!finalRuleFound) {
            console.warn("[getClashMetaConfig] No MATCH/FINAL rule found in remote config. Adding fallback: MATCH,DIRECT");
            ClashConfig.rules.push("MATCH,DIRECT");
        }
        console.info(`[getClashMetaConfig] Successfully processed ${ClashConfig.rules.length} rules.`);

    } catch (error) {
        console.error(`[getClashMetaConfig] Failed to process rules from ${Config.RemoteConfig}:`, error);
        console.warn("[getClashMetaConfig] Adding minimal fallback rule due to error.");
        if (!Array.isArray(ClashConfig.rules) || ClashConfig.rules.length === 0) {
            ClashConfig.rules = ["MATCH,DIRECT"];
        } else if (!ClashConfig.rules.some(rule => rule.startsWith("MATCH,"))) {
            ClashConfig.rules.push("MATCH,DIRECT");
        }
    }

    console.info("[getClashMetaConfig] Config generation completed.");
    return ClashConfig;
}

    isInsecure: true,

    isForcedRefresh: false

import { RemoteConfigReader } from "../RemoteConfigReader/main.js";

export async function getSingBoxConfig(
    Proxies,
    EdgeSubDB,
    PassedConfig = {}
) {
    const Config = TrulyAssign(BasicConfig, PassedConfig);

    let SingBoxConfig = JSON.parse(JSON.stringify(BasicSingBoxConfig));

    // Append rule sets
    let RemoteConfig = await (new RemoteConfigReader(Config.RemoteConfig)).Process(EdgeSubDB, Config.isForcedRefresh);

    // Update route rules
    SingBoxConfig.route.rules = RemoteConfig.RuleSet.map(ruleGroup => {
        const rule = {};
        rule.outbound = ruleGroup.Outbound;
        rule.Rules.forEach(ruleLine => {
            const parts = ruleLine.split(",");
            const key = parts[0].toLowerCase().replace(/-/g, "_");
            if (key === "final" || key === "match") {
                SingBoxConfig.route.final = ruleGroup.Outbound;
            } else {
                if (!rule[key]) rule[key] = [];
                rule[key].push(parts[1]);
            }
        });
        return rule;
    });

    // Append proxy groups
    for (const group of RemoteConfig.ProxyGroup) {
        const matchedProxies = Proxies.filter(proxy => 
            group.RegExps.some(reg => proxy.__Remark.match(new RegExp(reg)))
        );

        const proxiesList = [
            ...group.GroupSelectors.map(t => t.replace(/^\[\]/, "")),
            ...matchedProxies.map(p => p.__Remark)
        ];

        const proxyGroup = {
            tag: group.name,
            type: group.type === "select" ? "selector" : group.type,
            outbounds: proxiesList
        };

        if (["url-test", "load-balance", "fallback"].includes(group.type)) {
            proxyGroup.type = "urltest";
            proxyGroup.url = group.TestConfig.TestURL;
            proxyGroup.interval = `${group.TestConfig.Interval}s`;
            proxyGroup.tolerance = group.TestConfig.Tolerance;
        }

        SingBoxConfig.outbounds.push(proxyGroup);
    }

    // Append proxies
    const validProxies = Proxies
        .map(proxy => {
            if (Dumper.__validate(proxy)) {
                return Dumper[proxy.__Type](proxy);
            }
            return null;
        })
        .filter(proxy => proxy);

    SingBoxConfig.outbounds = [
        ...SingBoxConfig.outbounds,
        ...validProxies
    ];

    return SingBoxConfig;
}

function __genTransportWS (Obj) : TransportWS | undefined {
    const PassedTransportType = Obj.Query.obfs || Obj.Query.net || Obj.Query.type;

    }

    const host = Obj.Query.host || Obj.Query.obfsParam;
    const PathObj = new URL(`path:${Obj.Query.path}`)
    const ParsedMaxEarlyData = parseInt(Obj.Query.ed || PathObj.searchParams.get("ed"));
    
    const MaxEarlyData = isNaN(ParsedMaxEarlyData) ? undefined : ParsedMaxEarlyData;
    const EarlyDataHeaderName = Obj.Query.eh || (MaxEarlyData ? "Sec-WebSocket-Protocol" : undefined)

    return {
        path: PathObj.pathname,
        headers: host ? { host } : undefined,

        "early-data-header-name": EarlyDataHeaderName,
        //"v2ray-http-upgrade"?: unknown,

    }
}

function URLQueryWrapper (Obj) {
    let Childs = [];
    for (let i in Obj) {
        let value = Obj[i];

        }
    }
    return Childs.join("&");

export async function ini(RemoteConfigURL, CacheDB, isForcedRefresh) {
    console.info(`[ini Reader] Processing remote config: ${RemoteConfigURL}`);
    let Config = {
        RuleSet: [],
        ProxyGroup: []
    };

    let rawINI;
    try {
        rawINI = await fetchCached(RemoteConfigURL, "RemoteConfig", CacheDB, isForcedRefresh);
        if (!rawINI || typeof rawINI !== 'string') {
            throw new Error("Fetched remote config is empty or invalid.");
        }
        console.info(`[ini Reader] Raw INI content:`, rawINI);
    } catch (e) {
        console.error(`[ini Reader] Failed to fetch or read remote config: ${RemoteConfigURL}`, e);
        return Config;
    }

    const lines = rawINI.split('\n').map(line => line.trim()).filter(line => line && !line.startsWith(';'));
    let customProxyGroups = [];
    for (const line of lines) {
        if (line.startsWith("custom_proxy_group=")) {
            customProxyGroups.push(line.replace("custom_proxy_group=", ""));
        }
    }
    console.info(`[ini Reader] Found ${customProxyGroups.length} custom_proxy_group lines:`, customProxyGroups);

    if (customProxyGroups.length > 0) {
        console.info(`[ini Reader] Processing ${customProxyGroups.length} custom proxy groups.`);
        for (const groupString of customProxyGroups) {
            try {
                const ConfigArr = groupString.split("`").map(s => s.trim()).filter(Boolean);
                if (ConfigArr.length < 2) {
                    console.warn(`[ini Reader] Skipping malformed custom_proxy_group line (less than 2 parts): ${groupString}`);
                    continue;
                }

                const groupName = ConfigArr[0];
                const groupType = ConfigArr[1].toLowerCase();
                const args = ConfigArr.slice(2);

                let ProxyGroupConfig = {
                    name: groupName,
                    type: groupType,
                    RegExps: [],
                    GroupSelectors: [],
                    TestConfig: undefined
                };

                const testConfigTypes = ["url-test", "fallback", "load-balance"];
                let testUrl = null;
                let testParamsRaw = null;

                for (const arg of args) {
                    if (arg.startsWith("[]") || ["DIRECT", "REJECT"].includes(arg)) {
                        ProxyGroupConfig.GroupSelectors.push(arg);
                    } else if (testConfigTypes.includes(groupType) && (arg.startsWith("http://") || arg.startsWith("https://"))) {
                        if (testUrl === null) {
                            testUrl = arg;
                        } else {
                            console.warn(`[ini Reader] Found second URL "${arg}" in group "${groupName}". Treating as RegExp.`);
                            ProxyGroupConfig.RegExps.push(arg);
                        }
                    } else if (testConfigTypes.includes(groupType) && testUrl !== null && testParamsRaw === null) {
                        testParamsRaw = arg;
                    } else {
                        ProxyGroupConfig.RegExps.push(arg);
                    }
                }

                if (testConfigTypes.includes(groupType) && testUrl) {
                    ProxyGroupConfig.TestConfig = {
                        TestURL: testUrl,
                        Interval: 300,
                        Timeout: 5000,
                        Tolerance: 50
                    };
                    if (testParamsRaw) {
                        const testParams = testParamsRaw.split(',').map(p => p.trim());
                        const interval = parseInt(testParams[0], 10);
                        const timeout = parseInt(testParams[1], 10);
                        const tolerance = parseInt(testParams[2], 10);

                        if (!isNaN(interval)) ProxyGroupConfig.TestConfig.Interval = interval;
                        if (!isNaN(timeout)) ProxyGroupConfig.TestConfig.Timeout = timeout;
                        if (!isNaN(tolerance)) ProxyGroupConfig.TestConfig.Tolerance = tolerance;
                    }
                    console.info(`[ini Reader] Parsed TestConfig for group "${groupName}":`, ProxyGroupConfig.TestConfig);
                } else if (testConfigTypes.includes(groupType) && !testUrl) {
                    console.warn(`[ini Reader] Group "${groupName}" is type "${groupType}" but no test URL was found.`);
                }

                console.info(`[ini Reader] Parsed group "${groupName}": Type=${groupType}, RegExps=${JSON.stringify(ProxyGroupConfig.RegExps)}, Selectors=${JSON.stringify(ProxyGroupConfig.GroupSelectors)}`);
                Config.ProxyGroup.push(ProxyGroupConfig);

            } catch (groupError) {
                console.error(`[ini Reader] Failed to process custom_proxy_group line: ${groupString}`, groupError);
            }
        }
    } else {
        console.info("[ini Reader] No custom_proxy_group entries found.");
    }

    console.info(`[ini Reader] Finished processing. Returning ${Config.ProxyGroup.length} proxy groups.`);
    return Config;
}

    constructor (RuleSet) {

        this.RuleSet = { Outbound, Args };

        const { Outbound, Args } = this.RuleSet;
        let Rules = [];
        if (Args[0].startsWith("[]")) { // quote a proxy group when starts with `[]`
            Rules = [Args.join(",").replace(/^\[\]/, "")];
        } else {
            // this.RuleSet.Interval = Args[1]; // we are not using this for now, but it's a reminder #todo

            // get rule type, assume it's surge type when unmatched.
            let RuleType = Args[0].split(":");

            // get rule list
            let RawRuleURL = Args[0].replace(new RegExp(`^${RuleType}:`, ""))
            let RawRule = await fetchCached(RawRuleURL, "RuleSet", CacheDB, isForcedRefresh);

            // process rule list 
            Rules = await this.Parser[RuleType](RawRule)
        }
        return { Outbound, Rules }
    }
    Parser = {
        surge: RawRule => RawRule.replaceAll("\r", "\n").split("\n").filter(i => i && !i.startsWith("#")).map(i => i.trim())
    }

export async function fetchCached (URL, Prefix = "Cached", CacheDB, isForcedRefresh = false) {
    // const CacheKey = `RemoteConfig_${URL}`;
    const CacheKey = `${Prefix}_${URL}`;
    const CacheTTL =  6 * 60 * 60; // 6hrs for default

    // check if data was cached
    let CachedData;
    if (!CacheDB) {
        CachedData = null;
        console.info(`[fetchCached] - CacheDB is not configured, fetching.`)
    } else {
        if (isForcedRefresh) {
            CachedData = null;
        } else {
            CachedData = await CacheDB.get(CacheKey)
        }
    }

    // return cached data when it exist
    if (CachedData) {
        return CachedData;
    }

    console.info(`[fetchCached] - "${CacheKey}" is not cached, fetching.`)
    // only when CachedData == falsy
    let data = await fetch (URL)
        .then(res => {
            if (res.status === 200 || res.status === 304) {
                return res
            } else {
                throw "Failed to fetch."
            }
        })
        .then(res => res.text())
        .catch(() => false);
    // save it into CacheDB when success, and, whatever, when CacheDB exist.
    if (data !== false && CacheDB) {
        await CacheDB.put(CacheKey, data, { expirationTtl: CacheTTL })
    }
    // just return
    return data;
}

export function removeUndefinedFields (obj: object) : object {
    for (const key in obj) {
        if (typeof obj[key] === 'object' && obj[key] !== null) {
            removeUndefinedFields(obj[key]);
        }
        if (obj[key] === undefined) {
            delete obj[key];
        }
    }
    return obj;
}

/**
 * Assign source to target, note that its only for the first frame. 
 * You will not get the target modified, instead, the target you need is in the returns.
 * @param {Object} target 
 * @param {Object} source 
 * @returns {Object} Assigned Target
 */
export function TrulyAssign (target, source) {
    let DeepCopiedTarget = JSON.parse(JSON.stringify(target))
    for (let key in source) {
        if (key in source && source[key] !== "__DEFAULT") {
            DeepCopiedTarget[key] = source[key];
        }
    }
    return DeepCopiedTarget;
}

        return JSON.parse(jsonString || "{}");
    } catch (e) {
        console.warn(`[safeJsonParse] Error parsing JSON: ${e.message}. Falling back to default.`);
        return defaultValue;
    }
}


export async function onRequest(context, isClashOriginal = false) {
    const { request, env } = context; // Destructure env for EdgeSubDB
    const URLObject = new URL(request.url);

    try {
        // --- 1. Parse Parameters ---
        const urlParam = URLObject.searchParams.get("url");
        const showHostParam = URLObject.searchParams.get("show_host") === "true";
        const httpHeadersParam = URLObject.searchParams.get("http_headers");
        const udpParam = URLObject.searchParams.get("udp") === "true";
        const ssUotParam = URLObject.searchParams.get("ss_uot") === "true";
        const remoteConfigParam = URLObject.searchParams.get("remote_config") || undefined; // Use undefined to trigger default in getClashMetaConfig
        const forcedRefreshParam = URLObject.searchParams.get("forced_refresh") === "true";
        const forcedWs0rttParam = URLObject.searchParams.get("forced_ws0rtt") === "true";

        if (!urlParam) {
            return new Response("Error: 'url' parameter is missing.", {
                status: 400,
                headers: { "Content-Type": "text/plain; charset=utf-8" }
            });
        }

        const httpHeaders = safeJsonParse(httpHeadersParam);

        // --- 2. Fetch and Parse Subscription Data ---
        console.info("[onRequest] Fetching and parsing subscription data...");
        let Proxies = await getParsedSubData(
            urlParam,
            env.EdgeSubDB, // Use env.EdgeSubDB
            showHostParam,
            httpHeaders
        );
        console.info(`[onRequest] Found ${Proxies.length} initial proxies.`);

        // --- 3. Filter Proxies (if Clash Original) ---
        if (isClashOriginal === true) {
            console.info("[onRequest] Filtering proxies for Clash Original compatibility.");
            const ClashSupportedProxyType = ["trojan", "vmess", "ss", "ssr", "http", "socks5"];
            const ClashSupportedSSCipher = [
                "aes-128-gcm", "aes-192-gcm", "aes-256-gcm",
                "chacha20-ietf-poly1305", "xchacha20-ietf-poly1305",
                "aes-128-cfb", "aes-192-cfb", "aes-256-cfb",
                "rc4-md5", "chacha20-ietf", "xchacha20",
                "aes-128-ctr", "aes-192-ctr", "aes-256-ctr"
            ];

            const initialProxyCount = Proxies.length;
            Proxies = Proxies
                .filter(proxy => proxy && ClashSupportedProxyType.includes(proxy.__Type)) // Ensure proxy exists and type is supported
                .filter(proxy => {
                    if (proxy.__Type === "ss") {
                        // Check if Auth and cipher exist and cipher is supported
                        const isSupported = proxy.Auth && ClashSupportedSSCipher.includes(proxy.Auth.cipher);
                        if (!isSupported) {
                            // *** Corrected console.warn syntax ***
                            console.warn(`[onRequest] Filtering out SS proxy "${proxy.__Remark || 'Unnamed'}" due to unsupported cipher: ${proxy.Auth?.cipher || 'N/A'}`);
                        }
                        return isSupported;
                    }
                    return true; // Keep non-SS proxies that passed the first filter
                });
            console.info(`[onRequest] Filtered proxies: ${initialProxyCount} -> ${Proxies.length}`);
        }

        // --- 4. Generate Clash Meta Config Object ---
        console.info("[onRequest] Generating Clash Meta config object...");
        let ClashMetaConfigObject = await getClashMetaConfig(
            Proxies,
            env.EdgeSubDB, // Use env.EdgeSubDB
            {
                isUDP: udpParam,
                isSSUoT: ssUotParam,
                isInsecure: true, // Assuming always true based on previous code
                RemoteConfig: remoteConfigParam,
                isForcedRefresh: forcedRefreshParam
            }
        );
        console.info("[onRequest] Config object generated.");

        // --- 5. Apply Forced WS 0-RTT (if requested) ---
        if (forcedWs0rttParam) {
            console.info("[onRequest] Applying Forced WS 0-RTT...");
            if (ClashMetaConfigObject && Array.isArray(ClashMetaConfigObject.proxies)) {
                let appliedCount = 0;
                for (let proxy of ClashMetaConfigObject.proxies) {
                    // Check if proxy is valid, has ws-opts as a valid object
                    if (proxy && typeof proxy === 'object' && proxy.hasOwnProperty('ws-opts') && typeof proxy['ws-opts'] === 'object' && proxy['ws-opts'] !== null) {
                        proxy["ws-opts"]["max-early-data"] = 2048; // Standard value
                        proxy["ws-opts"]["early-data-header-name"] = "Sec-WebSocket-Protocol";
                        appliedCount++;
                    } else if (proxy && typeof proxy === 'object' && (proxy.network === 'ws' || (proxy.transport && proxy.transport.type === 'ws'))) {
                        // *** Corrected console.warn syntax ***
                        console.warn(`[onRequest] Proxy "${proxy.name || 'Unnamed'}" uses WS but 'ws-opts' is missing/invalid. Cannot apply forced_ws0rtt.`);
                    }
                }
                console.info(`[onRequest] Applied Forced WS 0-RTT to ${appliedCount} proxies.`);
            } else {
                console.warn("[onRequest] Cannot apply forced_ws0rtt: ClashMetaConfigObject or its proxies array is invalid.");
            }
        }

         // --- 6. Dump Config to YAML ---
        console.info("[onRequest] Dumping config object to YAML...");
        let ResponseBody;
        try {
            if (!ClashMetaConfigObject || typeof ClashMetaConfigObject !== 'object') {
                throw new Error("Generated config object is invalid or empty.");
            }

            console.log("[onRequest] Final proxy-groups before YAML dump:", JSON.stringify(ClashMetaConfigObject["proxy-groups"], null, 2));

            ResponseBody = Yaml.dump(ClashMetaConfigObject);
            console.info("[onRequest] YAML dump successful.");
        } catch (e) {
            const errorMessage = `Error: Failed to generate YAML config. Internal error: ${e.message}`;
            return new Response(errorMessage, {
                status: 500,
                headers: { "Content-Type": "text/plain; charset=utf-8" }
            });
        }
        // --- 7. Return Response ---
        console.info("[onRequest] Sending YAML response.");
        // *** Use TextEncoder for Content-Length in Worker environments ***
        const responseHeaders = {
            "Content-Type": "text/plain; charset=utf-8",
            "Content-Length": new TextEncoder().encode(ResponseBody).length.toString()
        };
        return new Response(ResponseBody, {
            status: 200,
            headers: responseHeaders
        });

    } catch (e) {
        // --- Global Error Handling ---
        console.error("[onRequest] Unexpected error:", e.message, e.stack);
        // *** Corrected Response generation in catch block ***
        const errorMessage = `Error: An unexpected error occurred. ${e.message}`;
        return new Response(errorMessage, {
            status: 500,
            headers: { "Content-Type": "text/plain; charset=utf-8" }
        });
    }
}

import getParsedSubData from "../internal/getParsedSubData.ts";
import { ShareLinkDumper } from "../internal/Dumpers/share-link.js";

export async function onRequest (context, isBase64 = false) {
    const { request } = context;
    const URLObject = new URL(request.url);
    // do convert
    const Proxies = await getParsedSubData(
        URLObject.searchParams.get("url"), 
        context.env.EdgeSubDB, 
        URLObject.searchParams.get("show_host") === "true",
        JSON.parse(URLObject.searchParams.get("http_headers")),
    );
    let Dumper = new ShareLinkDumper();
    let ShareLinkArray = [];
    for (let i of Proxies) {
        if (Dumper[i.__Type]) {
            ShareLinkArray.push(Dumper[i.__Type](i))
        }
    }
    
    // generate final response
    let ShareLinkResponse = ShareLinkArray.join("\n");
    if (isBase64 === true) {
        ShareLinkResponse = btoa(ShareLinkResponse);
    }

    return new Response(ShareLinkResponse, {
        status: 200,
        headers: {
            "Content-Type": "text/plain, charset=utf-8",
            "Content-Length": ShareLinkResponse.length
        }
    })
}

import { getSingBoxConfig } from "../internal/Converter/getSingBoxConfig.js";
import getParsedSubData from "../internal/getParsedSubData.ts";

export async function onRequest (context) {
    const { request } = context;
    const URLObject = new URL(request.url);
    let Proxies = await getParsedSubData(
        URLObject.searchParams.get("url"), 
        context.env.EdgeSubDB, 
        URLObject.searchParams.get("show_host") === "true",
        JSON.parse(URLObject.searchParams.get("http_headers")),
    );

    // a javascript object !!! not YAML !!!
    let SingBoxConfigObject = await getSingBoxConfig (
        Proxies,
        context.env.EdgeSubDB,
        {
            isUDP: URLObject.searchParams.get("udp") === "true",
            isSSUoT: URLObject.searchParams.get("ss_uot") === "true",
            isInsecure: true,
            RemoteConfig: URLObject.searchParams.get("remote_config") || "__DEFAULT",
            isForcedRefresh: URLObject.searchParams.get("forced_refresh") === "true" ? true : false
        }
    )

    // handle forced ws 0-rtt
    if (URLObject.searchParams.get("forced_ws0rtt") === "true") {
        console.info("[Main] ForcedWS0RTT enabled.")
        for (let i of SingBoxConfigObject.outbounds) {
            if (!("transport" in i)) {
                continue;
            }
            if (i.transport.type !== "ws") {
                continue;
            }
            i.transport.max_early_data = 2560
            i.transport.early_data_header_name = "Sec-WebSocket-Protocol"
        }
    }

    const ResponseBody = JSON.stringify(SingBoxConfigObject)

    return new Response(ResponseBody, {
        status: 200,
        headers: {
            "Content-Type": "application/json; charset=utf-8",
            "Content-Length": ResponseBody.length,
        }
    })
}
